const db = require('../../config/db.js');

async function getFirstRootAdmin() {
    const rows = await db.query(`
        SELECT u.user_id, u.username, u.email, r.name AS role_name, r.role_id
        FROM users u
        JOIN roles r ON u.role_id = r.role_id
        WHERE LOWER(u.username) = LOWER(?)
        ORDER BY u.user_id ASC
        LIMIT 1
    `, ['admin']);
    return rows[0];
}

async function hasOwnedProjects(userId) {
    const rows = await db.query(
        'SELECT project_id FROM projects WHERE contact_person_id = ? LIMIT 1',
        [Number(userId)]
    );
    return Array.isArray(rows) && rows.length > 0;
}

/**
 * Protokolliert eine Admin-Aktion in `admin_actions_log`.
 * @async
 * @param {number|null} actorId - ID des ausführenden Admins.
 * @param {number|null} targetId - ID des betroffenen Benutzers.
 * @param {string} action - Aktionstyp.
 * @param {string|null} [details=null] - Optionale Detailinformation.
 * @param {object} [conn=db] - Optionaler DB-Connector/Transaktions-Conn.
 * @returns {Promise<void>}
 */
async function logAdminAction(actorId, targetId, action, details = null, conn = db) {
    try {
        await conn.query(
            'INSERT INTO admin_actions_log (actor_id, target_user_id, action, details) VALUES (?, ?, ?, ?)',
            [actorId, targetId || null, action, details]
        );
    } catch (err) {
        console.warn('Log-Fehler Admin-Aktion:', err.message);
    }
}

/**
 * Liefert Admin-Logs (neueste zuerst).
 * @async
 * @param {number} [limit=100]
 * @returns {Promise<Array<Object>>}
 */
async function getAdminLogs(limit = 100) {
    const sql = `
        SELECT l.*, u.username as actor_username, t.username as target_username
        FROM admin_actions_log l
        LEFT JOIN users u ON l.actor_id = u.user_id
        LEFT JOIN users t ON l.target_user_id = t.user_id
        ORDER BY l.created_at DESC
        LIMIT ?
    `;
    return await db.query(sql, [limit]);
}

/**
 * Aktualisiert die Rolle eines Benutzers und protokolliert die Aktion.
 * @async
 * @param {number} adminUserId
 * @param {number} targetUserId
 * @param {number} newRoleId
 * @returns {Promise<void>}
 */
async function updateUserRole(adminUserId, targetUserId, newRoleId) {
    const adminIdNum = Number(adminUserId);
    const newRoleIdNum = Number(newRoleId);
    const targetIdNum = Number(targetUserId);

    const actorUser = await getUserForAdmin(adminIdNum);
    if (!actorUser) {
        throw new Error('FORBIDDEN: Ausführender Admin nicht gefunden');
    }

    const targetUser = await getUserForAdmin(targetIdNum);
    if (!targetUser) {
        throw new Error('BAD_REQUEST: Zielbenutzer nicht gefunden');
    }

    const roleRows = await db.query('SELECT name FROM roles WHERE role_id = ? LIMIT 1', [newRoleIdNum]);
    const role = roleRows[0];
    if (!role) {
        throw new Error('BAD_REQUEST: Rolle nicht gefunden');
    }

    if (Number(targetUser.role_id) === newRoleIdNum) {
        return;
    }

    const rootAdmin = await getFirstRootAdmin();
    const actorIsRootAdmin = !!rootAdmin && Number(rootAdmin.user_id) === adminIdNum;
    const targetIsRootAdmin = !!rootAdmin && Number(rootAdmin.user_id) === targetIdNum;

    if (targetUser.role_name === 'Admin' && role.name !== 'Admin') {
        if (targetIsRootAdmin) {
            throw new Error('FORBIDDEN: Der Root-Admin kann nicht herabgestuft werden.');
        }
        if (!actorIsRootAdmin) {
            throw new Error('FORBIDDEN: Nur der Root-Admin darf andere Administratoren herabstufen.');
        }
    }

    // Users who still own at least one project cannot be downgraded to Mitarbeiter.
    if (role.name === 'Mitarbeiter' && targetUser.role_name !== 'Mitarbeiter') {
        if (await hasOwnedProjects(targetIdNum)) {
            throw new Error('BAD_REQUEST: Benutzer ist Ansprechpartner in mindestens einem Projekt und kann nicht zu Mitarbeiter degradiert werden.');
        }
    }

    await db.query('UPDATE users SET role_id = ? WHERE user_id = ?', [newRoleIdNum, targetIdNum]);
    await logAdminAction(adminUserId, targetUserId, 'ROLE_CHANGE', `New Role ID: ${newRoleIdNum}`);
}

/**
 * Befördert einen Benutzer auf eine Zielrolle anhand des Rollennamens.
 * Wenn die Zielrolle bereits gesetzt ist, wird keine Änderung ausgeführt.
 * @async
 * @param {number|string} targetUserId
 * @param {string} roleName
 * @param {number|string|null} actorId
 * @param {object} [conn=db]
 * @returns {Promise<{updated: boolean, roleId: number}>}
 */
async function promoteToRole(targetUserId, roleName, actorId, conn = db) {
    const roleRows = await conn.query('SELECT role_id FROM roles WHERE name = ? LIMIT 1', [roleName]);
    const role = roleRows[0];
    if (!role) {
        throw new Error(`BAD_REQUEST: Rolle nicht gefunden: ${roleName}`);
    }

    const userRows = await conn.query('SELECT user_id, role_id FROM users WHERE user_id = ? LIMIT 1', [targetUserId]);
    const user = userRows[0];
    if (!user) {
        throw new Error('BAD_REQUEST: Ansprechpartner nicht gefunden');
    }

    if (Number(user.role_id) === Number(role.role_id)) {
        return { updated: false, roleId: Number(role.role_id) };
    }

    await conn.query('UPDATE users SET role_id = ? WHERE user_id = ?', [role.role_id, targetUserId]);
    await logAdminAction(actorId || null, targetUserId, 'role_promote', `auto-promote to ${roleName}`, conn);
    return { updated: true, roleId: Number(role.role_id) };
}

async function deleteUser(adminUserId, targetUserId) {

}

/**
 * Vergibt manuell Punkte an einen Benutzer (Admin-Funktion).
 * @async
 * @param {number} adminUserId
 * @param {number} targetUserId
 * @param {number} points
 * @param {string} reason
 * @returns {Promise<void>}
 */
async function assignManualPoints(adminUserId, targetUserId, points, reason) {
    await pointsService.addPendingDelta({ userId: targetUserId, delta: points, reason, source: 'admin-manual', createdBy: adminUserId });
    await logAdminAction(adminUserId, targetUserId, 'manual-points', `points=${points}, reason=${reason}`);
}

/**
 * Setzt das Passwort eines Benutzers zurück (hashed value bereitgestellt) und protokolliert.
 * @async
 * @param {number} adminUserId
 * @param {number} targetUserId
 * @param {string} hashedValue
 * @returns {Promise<void>}
 */
async function resetUserPassword(adminUserId, targetUserId, hashedValue) {
    await db.query('UPDATE users SET password_hash = ? WHERE user_id = ?', [hashedValue, targetUserId]);
    await logAdminAction(adminUserId, targetUserId, 'reset-password', `generated_password=1`);
}

    const sql = `

        FROM users u
        JOIN roles r ON u.role_id = r.role_id
        WHERE u.user_id = ?
    `;
    const rows = await db.query(sql, [userId]);
    return rows[0];
}
